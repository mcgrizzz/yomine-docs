import { Card, CardGrid } from '@astrojs/starlight/components';

## Features


<CardGrid stagger>
	<Card title="Mine vocabulary from real content" icon="document">
		Analyze subtitle files to extract vocabulary directly from the shows,
		movies, and videos you actually watch.
	</Card>

	<Card title="Direct video navigation" icon="seti:video">
		Jump straight from a term to the exact moment in video using ASBPlayer or MPV
		integration. Quickly mine the term and move on to the next one.
	</Card>

	<Card title="Smart ranking & filtering" icon="setting">
		Prioritize words using flexible ranking and filtering across frequency,
		parts of speech, and sentence comprehension - or directly search for a
		specific term to find it in real context.
	</Card>

	<Card title="Anki Integration" icon="open-book">
		Automatically filter out terms you already have cards for. No need to
		manually check each term.
	</Card>

	<Card title="Ignore List Support" icon="moon">
		Create ignore lists for words you already know or have no interest in
		learning. For example, English loan words.
	</Card>
</CardGrid>
